  REAL(r64)                    :: CurveOutput             = 0.0d0
  REAL(r64)                    :: CurveInput1             = 0.0d0   !- MassFlow                                         [kg/s]
  REAL(r64)                    :: CurveInput2             = 0.0d0   !- Density                                          [kg/m3]
